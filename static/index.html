const TelegramBot = require('node-telegram-bot-api');
const { checkProductAvailability } = require('./scraper');
const cron = require('node-cron');
require('dotenv').config();

const bot = new TelegramBot(process.env.BOT_TOKEN, { polling: true });
const monitoredProducts = new Map();
const userSubscriptions = new Set();

// Start command
bot.onText(/\/start/, (msg) => {
    const chatId = msg.chat.id;
    userSubscriptions.add(chatId);
    
    bot.sendMessage(chatId, `
ü§ñ *Stock Monitor Bot Started!*

Available commands:
/monitor \\ - Start monitoring a product
/stop \\ - Stop monitoring a product  
/status - Show current monitoring status
/help - Show this help message

Example:
\`/monitor https://shop.amul.com/en/product/amul-high-protein-rose-lassi-200-ml-or-pack-of-30\`
    `, { parse_mode: 'Markdown' });
});

// Monitor command
bot.onText(/\/monitor (.+)/, async (msg, match) => {
    const chatId = msg.chat.id;
    const url = match[1];
    
    if (!isValidUrl(url)) {
        bot.sendMessage(chatId, '‚ùå Invalid URL format');
        return;
    }
    
    monitoredProducts.set(url, {
        chatIds: [chatId],
        lastStatus: null,
        addedAt: new Date()
    });
    
    bot.sendMessage(chatId, `‚úÖ Started monitoring: ${url}`);
    
    // Initial check
    try {
        const isAvailable = await checkProductAvailability(url);
        const status = isAvailable ? '‚úÖ In Stock' : '‚ùå Out of Stock';
        bot.sendMessage(chatId, `Current status: ${status}`);
    } catch (error) {
        bot.sendMessage(chatId, '‚ö†Ô∏è Error checking product status');
    }
});

// Stop monitoring command
bot.onText(/\/stop (.+)/, (msg, match) => {
    const chatId = msg.chat.id;
    const url = match[1];
    
    if (monitoredProducts.has(url)) {
        monitoredProducts.delete(url);
        bot.sendMessage(chatId, `üõë Stopped monitoring: ${url}`);
    } else {
        bot.sendMessage(chatId, '‚ùå Product not found in monitoring list');
    }
});

// Status command
bot.onText(/\/status/, (msg) => {
    const chatId = msg.chat.id;
    
    if (monitoredProducts.size === 0) {
        bot.sendMessage(chatId, 'üìã No products currently being monitored');
        return;
    }
    
    let statusMessage = 'üìã *Monitored Products:*\\n\\n';
    
    for (const [url, data] of monitoredProducts) {
        const status = data.lastStatus ? '‚úÖ In Stock' : '‚ùå Out of Stock';
        statusMessage += `‚Ä¢ ${url}\\n  Status: ${status}\\n\\n`;
    }
    
    bot.sendMessage(chatId, statusMessage, { parse_mode: 'Markdown' });
});

// Help command
bot.onText(/\/help/, (msg) => {
    const chatId = msg.chat.id;
    bot.sendMessage(chatId, `
ü§ñ *Telegram Stock Monitor Bot*

*Commands:*
/start - Initialize the bot
/monitor \\ - Add product to monitoring
/stop \\ - Remove product from monitoring
/status - Show monitoring status
/help - Show this help

*Example Usage:*
\`/monitor https://shop.amul.com/en/product/amul-high-protein-rose-lassi-200-ml-or-pack-of-30\`

The bot checks product availability every 10 minutes and notifies you when items come back in stock.
    `, { parse_mode: 'Markdown' });
});

function isValidUrl(string) {
    try {
        new URL(string);
        return true;
    } catch (_) {
        return false;
    }
}

module.exports = { bot, monitoredProducts, userSubscriptions };

Web Scraping Logic

const puppeteer = require('puppeteer');

async function checkProductAvailability(url) {
    let browser;
    
    try {
        browser = await puppeteer.launch({
            headless: true,
            args: [
                '--no-sandbox',
                '--disable-setuid-sandbox',
                '--disable-dev-shm-usage',
                '--disable-accelerated-2d-canvas',
                '--no-first-run',
                '--no-zygote',
                '--single-process',
                '--disable-gpu'
            ]
        });
        
        const page = await browser.newPage();
        
        // Set user agent to avoid bot detection
        await page.setUserAgent('Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36');
        
        // Set viewport
        await page.setViewport({ width: 1366, height: 768 });
        
        console.log(`Checking availability for: ${url}`);
        
        // Navigate to product page
        await page.goto(url, { 
            waitUntil: 'networkidle2',
            timeout: 30000 
        });
        
        // Wait for page to load completely
        await page.waitForTimeout(3000);
        
        // Check if pincode selection is needed
        const pincodeModal = await page.$('.pincode-modal, .location-modal');
        if (pincodeModal) {
            await handlePincodeSelection(page);
        }
        
        // Look for add to cart button
        const addToCartButton = await page.$(
            'button[type="submit"]:not([disabled]), ' +
            '.add-to-cart:not([disabled]), ' +
            '.btn-add-cart:not([disabled]), ' +
            'input[type="submit"][value*="cart"]:not([disabled])'
        );
        
        if (!addToCartButton) {
            // Check for out of stock indicators
            const outOfStockElements = await page.$$eval(
                '.out-of-stock, .sold-out, .unavailable, .notify-me',
                elements => elements.length > 0
            );
            
            return !outOfStockElements;
        }
        
        // Check if button is disabled
        const isDisabled = await page.evaluate((button) => {
            return button.disabled || 
                   button.classList.contains('disabled') ||
                   button.getAttribute('aria-disabled') === 'true';
        }, addToCartButton);
        
        const buttonText = await page.evaluate((button) => {
            return button.textContent || button.value || '';
        }, addToCartButton);
        
        console.log(`Button text: ${buttonText}, Disabled: ${isDisabled}`);
        
        // Button is available if it's not disabled and doesn't contain "notify" text
        const isAvailable = !isDisabled && 
                           !buttonText.toLowerCase().includes('notify') &&
                           !buttonText.toLowerCase().includes('sold out') &&
                           !buttonText.toLowerCase().includes('unavailable');
        
        return isAvailable;
        
    } catch (error) {
        console.error('Error checking product availability:', error);
        throw error;
    } finally {
        if (browser) {
            await browser.close();
        }
    }
}

async function handlePincodeSelection(page) {
    try {
        // Enter pincode if modal is present
        const pincodeInput = await page.$('input[name="pincode"], input[placeholder*="pincode"], .pincode-input');
        
        if (pincodeInput) {
            await pincodeInput.type(process.env.PINCODE || '400001');
            
            // Click submit/check button
            const submitBtn = await page.$('.check-pincode, .submit-pincode, button[type="submit"]');
            if (submitBtn) {
                await submitBtn.click();
                await page.waitForTimeout(2000);
            }
        }
        
        // Close modal if still open
        const closeBtn = await page.$('.close, .modal-close, .dismiss');
        if (closeBtn) {
            await closeBtn.click();
            await page.waitForTimeout(1000);
        }
        
    } catch (error) {
        console.log('Error handling pincode selection:', error);
    }
}

module.exports = {
    checkProductAvailability
};

Scheduling System

const cron = require('node-cron');
const { checkProductAvailability } = require('./scraper');
const { bot, monitoredProducts } = require('./bot');

// Check products every 10 minutes
const task = cron.schedule(process.env.CHECK_INTERVAL || '*/10 * * * *', async () => {
    console.log('üîÑ Running scheduled availability check...');
    
    if (monitoredProducts.size === 0) {
        console.log('No products to monitor');
        return;
    }
    
    const checkPromises = Array.from(monitoredProducts.entries()).map(
        ([url, productData]) => checkSingleProduct(url, productData)
    );
    
    try {
        await Promise.all(checkPromises);
        console.log('‚úÖ Completed all product checks');
    } catch (error) {
        console.error('‚ùå Error during scheduled check:', error);
    }
}, {
    scheduled: false // Don't start immediately
});

async function checkSingleProduct(url, productData) {
    try {
        console.log(`Checking: ${url}`);
        
        const isAvailable = await checkProductAvailability(url);
        const wasAvailable = productData.lastStatus;
        
        // Update status
        productData.lastStatus = isAvailable;
        
        // Send notification if status changed from unavailable to available
        if (!wasAvailable && isAvailable) {
            await notifyUsers(url, productData.chatIds);
        }
        
        console.log(`${url}: ${isAvailable ? '‚úÖ Available' : '‚ùå Unavailable'}`);
        
    } catch (error) {
        console.error(`Error checking ${url}:`, error);
        
        // Notify users about the error
        const errorMessage = `‚ö†Ô∏è Error checking product availability:
${url}

The bot will continue monitoring and try again on the next scheduled check.`;
        
        productData.chatIds.forEach(chatId => {
            bot.sendMessage(chatId, errorMessage);
        });
    }
}

async function notifyUsers(url, chatIds) {
    const message = `üéâ *PRODUCT AVAILABLE!* üéâ

The product you were monitoring is now in stock:

üîó ${url}

üèÉ‚Äç‚ôÇÔ∏è *Quick! Go grab it before it's out of stock again!*

Use /stop ${url} to stop monitoring this product.`;
    
    const notificationPromises = chatIds.map(chatId => 
        bot.sendMessage(chatId, message, { 
            parse_mode: 'Markdown',
            disable_web_page_preview: false
        }).catch(err => {
            console.error(`Failed to send notification to ${chatId}:`, err);
        })
    );
    
    await Promise.all(notificationPromises);
    console.log(`üì¢ Sent notifications to ${chatIds.length} users`);
}

// Function to start the scheduler
function startScheduler() {
    task.start();
    console.log('üìÖ Product availability scheduler started');
    console.log(`‚è∞ Check interval: ${process.env.CHECK_INTERVAL || '*/10 * * * *'}`);
}

// Function to stop the scheduler
function stopScheduler() {
    task.stop();
    console.log('‚èπÔ∏è Product availability scheduler stopped');
}

// Manual check function for testing
async function runManualCheck() {
    console.log('üîÑ Running manual check...');
    
    for (const [url, productData] of monitoredProducts) {
        await checkSingleProduct(url, productData);

Notification System

const { bot } = require('./bot');

class NotificationManager {
    constructor() {
        this.rateLimits = new Map();
        this.messageQueue = [];
        this.isProcessingQueue = false;
    }
    
    async sendStockAlert(chatId, productUrl) {
        if (this.isRateLimited(chatId)) {
            console.log(`Rate limited notification for user ${chatId}`);
            return;
        }
        
        const message = this.formatStockAlertMessage(productUrl);
        
        try {
            await bot.sendMessage(chatId, message, {
                parse_mode: 'Markdown',
                disable_web_page_preview: false,
                reply_markup: {
                    inline_keyboard: [[
                        {
                            text: 'üõí Buy Now',
                            url: productUrl
                        },
                        {
                            text: 'üõë Stop Monitoring',
                            callback_data: `stop_${btoa(productUrl)}`
                        }
                    ]]
                }
            });
            
            this.updateRateLimit(chatId);
            console.log(`‚úÖ Stock alert sent to ${chatId}`);
            
        } catch (error) {
            console.error(`‚ùå Failed to send stock alert to ${chatId}:`, error);
            
            if (error.response && error.response.statusCode === 429) {
                this.queueMessage(chatId, message);
            }
        }
    }
    
    formatStockAlertMessage(productUrl) {
        const productName = this.extractProductName(productUrl);
        
        return `üéâ *STOCK ALERT!* üéâ

üì¶ *${productName}* is now available!

üîó ${productUrl}

‚ö° *Act fast!* Popular items sell out quickly.

Use /stop to stop monitoring this product.`;
    }
    
    extractProductName(url) {
        try {
            const urlParts = url.split('/');
            const productSlug = urlParts[urlParts.length - 1];
            return productSlug
                .replace(/-/g, ' ')
                .replace(/\b\w/g, l => l.toUpperCase());
        } catch {
            return 'Product';
        }
    }
    
    isRateLimited(chatId) {
        const lastSent = this.rateLimits.get(chatId);
        if (!lastSent) return false;
        
        const timeDiff = Date.now() - lastSent;
        const minInterval = 60000; // 1 minute minimum between notifications
        
        return timeDiff < minInterval;
    }
    
    updateRateLimit(chatId) {
        this.rateLimits.set(chatId, Date.now());
    }
    
    queueMessage(chatId, message) {
        this.messageQueue.push({
            chatId,
            message,
            timestamp: Date.now()
        });
        
        if (!this.isProcessingQueue) {
            this.processMessageQueue();
        }
    }
    
    async processMessageQueue() {
        this.isProcessingQueue = true;
        
        while (this.messageQueue.length > 0) {
            const queuedMessage = this.messageQueue.shift();
            
            try {
                await bot.sendMessage(queuedMessage.chatId, queuedMessage.message);
                console.log(`‚úÖ Queued message sent to ${queuedMessage.chatId}`);
            } catch (error) {
                console.error(`‚ùå Failed to send queued message:`, error);
            }
            
            await new Promise(resolve => setTimeout(resolve, 1000));
        }
        
        this.isProcessingQueue = false;
    }
}

module.exports = {
    NotificationManager
};
